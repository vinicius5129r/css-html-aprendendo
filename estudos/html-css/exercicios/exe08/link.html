        Seletores Pesonalizados
        #  = id - Ex: #meuId
        .  = class - Ex: .minhaClasse
        :  = pseudo-classe - Ex: a:hover
        :: = pseudo-elemento - Ex: a::after
        >  = filho direto - Ex: div > p
        +  = irmão adjacente - Ex: h1 + p
